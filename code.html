                        function getValues()

                            //1. get value to test Fizz

                             displayData(fbData);
                        
                            }else{

                        function fizzBuzz(fizzValue,buzzValue){
                        
                        let returnArray =[];
                        
                        for (let index = 1; index < 100; index++) {
                           
                            if (index % fizzValue == 0 && index % buzzValue == 0)  {
                        
                                 returnArray.push('FizzBuzz');
                                
                            } else if (index % fizzValue == 0) {
                        
                                returnArray.push('Fizz');
                                
                            }
                        
                            else if(index % buzzValue == 0){
                                returnArray.push('Buzz');
                            }
                        
                            else{
                                returnArray.push(index);
                            }
                            
                        }
                        
                        return returnArray;
                        
                        }
                        
                        
                        
                        function displayData(fbData){
                        
                         let templateRows = "";
                        
                        
                         for (let index = 0; index < fbData.length; index++) {
                        
                                let fizzBuzzResult = fbData[index];
                            
                                let fizzColour = "primary"
                         
                            if (fizzBuzzResult == "Fizz") {
                                fizzColour = "green"
                                
                            } else if(fizzBuzzResult == "Buzz") {
                                fizzColour = "purple"
                            }else if(fizzBuzzResult == "FizzBuzz"){
                                fizzColour = "red"
                            }else{
                                fizzColour = "primary"
                            }
                        
                            templateRows += `${fizzBuzzResult}`;
                        
                            
                         }
                        
                         document.getElementById("results").innerHTML = templateRows;
                        
                        }

The Code is structured in three functions.

getValues - Function

This function allows us to grab the fizz and buzz values from the user on the app page using grab element by ID. The function also checks to validate that integers are being entered.

fizzBuzz - Function

This function takes the fizz and buzz values then using a for loop it iterates from start to end of 100 all while dividing using the modulus of the fiiz and buzz values if any value is divisiable by the fizz and buzz values then it will be stored as fizz or buzz if can be divised by both then it is stored as fizzbuzz.

displayData - Function

The display numbers uses a for loop to iterate through the array of generated numbers to add table rows to the resutls table in the app page. IF functions check the outcome and depenidng on fizz, buzz or fizzbuzz then the class colour will be altered using a css function.
